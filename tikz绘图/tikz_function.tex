\begin{tikzpicture}[scale = 1,thick]  % 统一箭头样式和线宽
    % 定义线段两端点（可任意修改）
    \coordinate (A) at  (-1.5, -0.3) ;    % 起点
    \coordinate (B) at (2.7, 1.75);    % 终点

    % 1. 手动绘制坐标轴（原点在中央）
    % x轴（带箭头）
    \draw[-Triangle,name path=xaxis] (-2, 0) -- (4, 0) node[right] {$x$};

    
    % y轴（带箭头）
    \draw[-Triangle] (0, -2) -- (0, 3) node[above] {$y$};


    % 2. 绘制直线
    \draw[line width=1pt,,name path=linefun] (A) -- (B) node[right] {$l$};


    % 3. 手动提取坐标（纯TikZ语法，无函数定义）
    % 提取A点坐标到\Ax和\Ay
    \pgfpointanchor{A}{center}
    \pgfgetlastxy{\Ax}{\Ay}
    % 提取B点坐标到\Bx和\By
    \pgfpointanchor{B}{center}
    \pgfgetlastxy{\Bx}{\By}

    % 4. 手动计算倾斜角（分步写，不封装函数）
    \pgfmathsetmacro{\dx}{\Bx - \Ax}   % 计算Δx
    \pgfmathsetmacro{\dy}{\By - \Ay}   % 计算Δy
    \pgfmathsetmacro{\angleRadian}{atan2(\dy, \dx)}  % 弧度制角度
    % \pgfmathsetmacro{\angleDegree}{rtd(\angleRadian)} % 转换为角度制

    % 语法：name intersections={of=线段A and 线段B, by=交点名}
    \path[name intersections={of=linefun and xaxis, by=crossPoint}];


    \draw (crossPoint)++(5mm,0) arc[start angle=0, end angle=\angleRadian, radius=5mm];

\end{tikzpicture}
