\begin{tikzpicture}[scale=1]
    % 绘制坐标轴
    \draw[->, thick] (-2,0) -- (2,0) node[right] {$x$};
    \draw[->, thick] (0,-2) -- (0,2) node[above] {$y$};
    % 绘制函数图像，添加 thick 选项使线条变粗
    \draw[domain=-2:1, smooth, variable=\x, red, thick] plot ({\x},{1*\x + 1});

    \node[blue] at (1,1) {$y = 2x + 1$};
\end{tikzpicture}
